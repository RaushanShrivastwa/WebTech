/* ---- row_averages.js ---- */
function row_averages(matrix) {
  return matrix.map(function(row) {
    var sum = 0;
    for (var i = 0; i < row.length; i++) sum += row[i];
    return sum / row.length;
  });
}

/* ---- reverser.js ---- */
function reverser(n) {
  var isNeg = n < 0;
  var rev = String(Math.abs(n)).split('').reverse().join('');
  return (isNeg ? -1 : 1) * parseInt(rev, 10);
}

      if (gen === 'Senior Citizen (60+)') {
        if      (income <= 300000)  tax = 0;
        else if (income <= 500000)  tax = (income - 300000) * 0.05;

        else if (income <= 1000000) tax = 12500 + (income - 500000) * 0.20;
        else                         tax = 112500 + (income - 1000000) * 0.30;

      if (income <= 500000) tax = 0;

      var surcharge = 0;
      if      (income > 10000000) surcharge = tax * 0.15;

  <h2>6×6 Sudoku Solver</h2>
  <div id="grid"></div>
  <br>
  <button onclick="solvePuzzle()">Solve</button>
  <button onclick="clearPuzzle()">Clear</button>
  <p id="msg"></p>

  <script>
    // Initial puzzle (0 = empty)
    var initial = [
      [5,3,0,0,7,0],
      [6,0,0,1,9,0],
      [0,9,8,0,0,0],
      [8,0,0,0,6,0],
      [4,0,0,8,0,3],
      [7,0,0,0,2,0]
    ];

    // Build grid inputs
    var container = document.getElementById('grid');
    for (var r = 0; r < 6; r++) {
      for (var c = 0; c < 6; c++) {
        var inp = document.createElement('input');
        inp.maxLength = 1;
        inp.id = 'c' + r + c;
        if (initial[r][c]) {
          inp.value = initial[r][c];
          inp.readOnly = true;
        }
        container.appendChild(inp);
      }
    }

    function getGrid() {
      var g = [];
      for (var r = 0; r < 6; r++) {
        g[r] = [];
        for (var c = 0; c < 6; c++) {
          g[r][c] = parseInt(document.getElementById('c'+r+c).value) || 0;
        }
      }
      return g;
    }

    function setGrid(g) {
      for (var r = 0; r < 6; r++) {
        for (var c = 0; c < 6; c++) {
          var el = document.getElementById('c'+r+c);
          if (!el.readOnly) {
            el.value = g[r][c] || '';
            if (g[r][c]) el.className = 'solved';
          }
        }
      }
    }

    function isValid(g, r, c, n) {
      for (var i = 0; i < 6; i++) {
        if (g[r][i] === n || g[i][c] === n) return false;
      }
      // 2x3 box check
      var br = Math.floor(r / 2) * 2;
      var bc = Math.floor(c / 3) * 3;
      for (var i = 0; i < 2; i++) {
        for (var j = 0; j < 3; j++) {
          if (g[br+i][bc+j] === n) return false;
        }
      }
      return true;
    }

    function solve(g) {
      for (var r = 0; r < 6; r++) {
        for (var c = 0; c < 6; c++) {
          if (g[r][c] === 0) {
            for (var n = 1; n <= 6; n++) {
              if (isValid(g, r, c, n)) {
                g[r][c] = n;
                if (solve(g)) return true;
                g[r][c] = 0;
              }
            }
            return false;
          }
        }
      }
      return true;
    }

    function solvePuzzle() {
      var g = getGrid();
      if (solve(g)) {
        setGrid(g);
        document.getElementById('msg').textContent = '✔ Puzzle Solved!';
      } else {
        document.getElementById('msg').textContent = '✗ No solution found.';
      }
    }

    function clearPuzzle() {
      for (var r = 0; r < 6; r++) {
        for (var c = 0; c < 6; c++) {
          var el = document.getElementById('c'+r+c);
          if (!el.readOnly) { el.value = ''; el.className = ''; }
        }
      }
      document.getElementById('msg').textContent = '';
    }
  </script>
</body>
</html>

/* ---- counter.js ---- */
function counter(arr) {
  var neg = 0, zero = 0, pos = 0;
  for (var i = 0; i < arr.length; i++) {
    if      (arr[i] < 0) neg++;
    else if (arr[i] === 0) zero++;
    else                   pos++;
  }
  return { negative: neg, zero: zero, positive: pos };
}

/* Tests:
   counter([-3, 0, 5, -1, 0, 7]) → { negative:2, zero:2, positive:2 }
   counter([0, 0, 1, 2, -5])     → { negative:1, zero:2, positive:2 }
*/

// POST: Compute EMI and show result
app.post('/calculate', (req, res) => {
  const P = parseFloat(req.body.P);
  const n = parseInt(req.body.n);
  const r = parseFloat(req.body.r) / 100 / 12;   // monthly interest rate

  // EMI Formula: E = P * r * (1+r)^n / ((1+r)^n - 1)
  const E = P * r * Math.pow(1 + r, n) / (Math.pow(1 + r, n) - 1);
  const total    = E * n;
  const interest = total - P;

  res.send(`
    <!DOCTYPE html><html><body style="font-family:Arial;max-width:400px;margin:30px auto">
    <h2>EMI Result</h2>
    <table border="1" cellpadding="8" style="border-collapse:collapse;width:100%">
      <tr><td>Monthly EMI</td><td><b>$${E.toFixed(2)}</b></td></tr>
      <tr><td>Total Payment</td><td>$${total.toFixed(2)}</td></tr>
      <tr><td>Total Interest</td><td>$${interest.toFixed(2)}</td></tr>
    </table>
    <br><a href="/">← Back</a>
    </body></html>
  `);
});
